		void write(const char* fileName)
		{
			std::ofstream file(fileName);
			if (file.is_open())
			{
				file << this->data;

				file.close();
			}
		}

		String read(const char* fileName)
		{
			String str;
			std::ifstream file(fileName);
			if (file.is_open())
			{
				file>>data;

					str.data = strcpy(str.data, data);
					str.size = strlen(data);
				
				
				file.close();
			}
			return str;
		}
